public class GstValidation {
    public static boolean isValidGST(String gst) {
        if (gst == null || gst.length() != 15) return false;

        // First 2 should be digits (State Code)
        if (!Character.isDigit(gst.charAt(0)) || !Character.isDigit(gst.charAt(1))) return false;

        // Next 5 should be uppercase letters
        for (int i = 2; i < 7; i++) {
            if (!Character.isUpperCase(gst.charAt(i))) return false;
        }

        // Next 4 should be digits
        for (int i = 7; i < 11; i++) {
            if (!Character.isDigit(gst.charAt(i))) return false;
        }

        // 12th should be uppercase letter
        if (!Character.isUpperCase(gst.charAt(11))) return false;

        // 13th can be letter or digit
        char c13 = gst.charAt(12);
        if (!Character.isUpperCase(c13) && !Character.isDigit(c13)) return false;

        // 14th should be 'Z'
        if (gst.charAt(13) != 'Z') return false;

        // 15th can be letter or digit
        char c15 = gst.charAt(14);
        return Character.isUpperCase(c15) || Character.isDigit(c15);
    }

    public static void main(String[] args) {
        System.out.println(isValidGST("22AAAAA0000A1Z5")); // true
        System.out.println(isValidGST("22aaaaa0000A1Z5")); // false
    }
}
