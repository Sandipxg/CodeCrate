public class AadhaarValidation {
    public static boolean isValidAadhaar(String aadhaar) {
        if (aadhaar == null || aadhaar.length() != 12) return false;
        for (char c : aadhaar.toCharArray()) {
            if (!Character.isDigit(c)) return false;
        }
        return true;
    }

    public static void main(String[] args) {
        System.out.println(isValidAadhaar("123412341234")); // true
        System.out.println(isValidAadhaar("1234abcd1234")); // false
    }
}
