import java.util.*;

public class MissingNumberFinder {
    public static void main(String[] args) {
        int[] arr = {1, 2, 4, 5};
        Set<Integer> set = new HashSet<>();
        for (int num : arr) set.add(num);

        for (int i = 1; i <= 5; i++) {
            if (!set.contains(i)) System.out.println("Missing: " + i);
        }
    }
}
