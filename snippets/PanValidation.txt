public class PanValidation {
    public static boolean isValidPAN(String pan) {
        if (pan == null || pan.length() != 10) return false;

        for (int i = 0; i < 5; i++) {
            if (!Character.isUpperCase(pan.charAt(i))) return false;
        }
        for (int i = 5; i < 9; i++) {
            if (!Character.isDigit(pan.charAt(i))) return false;
        }
        return Character.isUpperCase(pan.charAt(9));
    }

    public static void main(String[] args) {
        System.out.println(isValidPAN("ABCDE1234F")); // true
        System.out.println(isValidPAN("abcde1234f")); // false
    }
}
